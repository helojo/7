#include <stdio.h>

int main(int argc, const char * argv[]) {
    int count=1;
    count++;
    count=5;
    int i=8;
    int y,b=3;
    y=(++b)+(b++);//8
    printf("%d\n",y);
    printf("%d %d %d %d %d\n",--i,i--,i,i++,++i);
    //printf("%d\n",i++);
    return 0;
}
