    // 1.pointer to pointer.cpp 

#include "stdafx.h"
#include <stdlib.h>

int _tmain(int argc, _TCHAR* argv[])
{
    int n = 0x88;
    int *pn = NULL;
    int **ppn = NULL;
    
    pn     = &n;
    ppn = &pn;
    printf("%X\r\n", ppn);
    printf("%X\r\n", *ppn);
    printf("%X\r\n", **ppn);

    system("pause");
    return 0;
}
