    //判断一个整数是否为素数（质数）
    //质数定义为在大于1的自然数中，除了1和它本身以外不再有其他因数，这样的数称为质数
    #include <stdio.h>
    int main()
    {
    int n, i, flag = 0;
    printf("请输入一个正整数：");
    scanf("%d", &n);
    for (i = 2; i <= n / 2; ++i)
    {
    //如果满足以下的条件，他就不是素数
    if (n%i == 0)
    {
    flag = 1;
    break;
    }
    }
    if (flag == 1)
    printf("%d不为素数", n);
    else
    printf("%d为素数", n);
    return 0;
    }
