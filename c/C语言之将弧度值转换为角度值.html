     1 #include<stdio.h>
 2 #include<stdlib.h>

 4 #define pi 3.141592

 6 void  conver_radian(float  radian)
 7 {
 8     int degree,minute,second;
 9     float angle,min,sec;
10 
11     angle = radian * 180 / pi;
12     degree = (int)angle;
13 
14     min = (angle - degree)*60;
15     minute = (int)min;
16 
17     sec = (min - minute)*60;
18     second = (int) (sec + 0.5);  //四舍五入
19 
20     printf("%f弧度等于%d度 %d分 %d秒\n",radian,degree,minute,second);
21 }
