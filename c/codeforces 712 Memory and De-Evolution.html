    #include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
    ll x, y;
    
    scanf("%lld %lld", &x, &y);
    ll maxx = y;
    ll midd = y;
    ll minn = y;
    ll p = 0;
    ll q = 0;
    ll count = 0;
    while(1)
    {
        p = maxx;
        q = midd;
        maxx = midd + maxx - 1;
        midd = p;
        minn = q;
        count++;
        if (maxx >= x)
        {
            cout << count + 2 << endl;
            return 0;
        }
    }    
    return 0;
}
