    指针的定义 ：var p *int  
    取指针的值 :*p


    ------------------------------------------------------------------------------------
    package main

     

    import (
        "fmt"
    )

     

    func main() {

     

        i := 0
        fmt.Println("i的地址：", &i)
        p := &i
        fmt.Printf("i对应的指针类型 %T,指向i的指针地址：%x", p, p)
        *p = 100
        fmt.Println("通过指针修改值100 i=", i)
    }
