     1     package main
 2 
 3 import "fmt"
 4 
 5 func main() {
 6     var sum int = 17
 7     var count int = 5
 8     var mean float64
 9 
10     mean = float64(count) / float64(sum)
11     fmt.Printf("mean 的值为: %f\n", mean)
12     fmt.Printf("mean 的值为: %.10f\n", float64(mean))
13     fmt.Printf("mean 的值为: %.20f\n", float64(mean))
14     fmt.Printf("mean 的值为: %.30f\n", float64(mean))
15     fmt.Printf("mean 的值为: %.40f\n", float64(mean))
16 
17     fmt.Printf("%e\n", 123400000.0)
18     fmt.Printf("%E\n", 123400000.0)
19 }
