    笛卡尔积二维数组的输出

    如下代码，简单易理解

     

    #include<iostream>
using namespace std;
 
int main()
{
    int a[100],b[100],c[2][100];
    int x,y,k;
    cin >> x;
    for(int i = 0;i < x;i++)
    cin >> a[i];

    cin >> y;
    for(int i = 0;i < y;i++)
    cin >> b[i];

    for(int i = 0;i < x;i++)
    {
        for(int j = 0;j < y;j++)
        {
            k = i * x + j;
            c[0][k] = a[i];
            c[1][k] = b[j];
            cout << "(" << c[0][k] << "," << c[1][k] << ")    ";
        }
        cout << endl;
    }
    return 0;
}
