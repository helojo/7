C++中continue使用

#include<iostream>
using namespace std;
int main()
{
    int value;
    int i;
    int sum=0;
    cout<<"input  value"<<endl;
    for(i=0;i<4;i++)
    {
        cin>>value;
        if(value<0)
        {
            cout<<"waring:resume load,value>0"<<endl;
            i--; 
            continue;
            /*
            continue跳过本次循环,回到循环条件测试部分,
            重新开始执行循环。
            */
        }
        sum+=value;
    }
    /*
    if(value==1)
    {
        cout<<"value=";
        cout<<value<<endl;
    }
    */
    cout<<"sum=";
    cout<<sum<<endl;
}
