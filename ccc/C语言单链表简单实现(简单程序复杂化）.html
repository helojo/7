void add(List* plist,int number){
    Node *p =(Node*)malloc(sizeof(Node));
    p->value =number;
    p->next =NULL;
    Node *last =plist->head;
    if(last){
        while(last->next){
            last = last->next;
        }
        last->next=p;
    }else{
        plist->head = p;
    }
}

void print(List* list){
    Node *p;
    for(p=list->head;p;p=p->next){
        printf("%d\t",p->value);
    }
    printf("\n");
}
