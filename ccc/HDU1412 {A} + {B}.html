    每个元素之间有一个空格隔开.

    每行最后一的元素后面没有空格，区别于HDU人见人爱A - B

    注意使用STL的时候要清空 。  a.clear();

    #include <bits/stdc++.h>
using namespace std;
set<int> a;
int b[100000];
int main()
{
    int n, m;
    int i, j;
    while (~scanf("%d %d", &n, &m))
    {
        for (i = 0; i < n + m; i++)
        {
            int x;
            scanf("%d", &x);
            a.insert(x);
        }
        // sort (a.begin(),a.end());
        int q = 0;
        memset(b,0,sizeof(b));
        for (set<int>::iterator it = a.begin() ; it != a.end(); it++)
        {
            b[q] = *it;
            q++;
        }
        for (int i = 0; i < q; i++)
        {
            if (i == q - 1)
            {
                cout << b[i];
            }
            else
            {
                cout << b[i] << " ";
            }
        }
        cout << endl;
        a.clear();
    }
    return 0;
}
