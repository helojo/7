#include <iostream>
#include <list>
#include <vector>
#include <string>
using namespace std;

void bubble(int*a,int n)//冒泡升序
{int j;
    for(int i=0;i<n;i++){
        for(int j=n-1;j>i;j--){
    
                if(a[j]<a[j-1])
                {
                    int temp=a[j];
                    a[j]=a[j-1];
                    a[j-1]=temp;
                    
                }
            
        }
    
}
}
int main(int argc, const char * argv[]) {
    // insert code here...

    int array[10]={1,2,3,23,21,13,11,-9,7,10};
    bubble(array, 10);
    for (int i=0;i<10; i++) {
        cout<<array[i]<<" ";
        
    }
    cout<<endl;
    return 0;
}
