 12 void creatList(LinkList &L,int n)

 14     L = (LinkList)malloc(sizeof(Lnode));

 16     LinkList rear = L;

 28 void traverse(LinkList L)

 44 LinkList mergeList(LinkList La,LinkList Lb,int n,int m)

 46     LinkList Lc,a,b,c;

 48     a = La->next;
 49     b = Lb->next;
