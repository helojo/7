//  1.指针变量只用于存储内存地址，根据内存地址去查找里面存储的数据。

#include <iostream>
using namespace std;
int main(int argc, const char * argv[]) {
    int a=15;
    int *p;
    int *q=&a;
    p=(int *)a;
    cout<<a<<endl;
    cout<<p<<endl;
    cout<<&a<<endl;
    cout<<q<<endl;
    return 0;
}
