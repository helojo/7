    #include<iostream>
using namespace std;
int fun(int x)
{
    int temp = x;
    int y = 0;
    while(temp)
    {
        int x1 = temp % 10;//5        
        temp = temp / 10;//43
        y = y * 10 + x1;//50
    }
    return y;
}

int main()
{
    int n,m;
    cin >> n >> m;
    int nn = fun(n);
    int mm = fun(m);
    int sum = nn + mm;
    int end = fun(sum);
    cout << end << endl;
    return 0;
}
