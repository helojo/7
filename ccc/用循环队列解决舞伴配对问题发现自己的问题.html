int emptyqueue(queue man) {//用于判断队列中还有没有人
    if (man.front == man.rear)
        return 0;
    else
        return 1;
}
int dequeue(queue &man, char *str) {//删除队首元素
    if (man.front == man.rear)
        return 0;
    else {
        strcpy(str, man.elem[man.front]);
        man.front = (man.front + 1) % 100;
        man.number--;
        return 1;
    }
}
