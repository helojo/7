    这里用了ｆｌｏａｔ类型

    公鸡x、母鸡y、小鸡z共100只

    钱：5ｘ +　３ｙ　＋　１／３ｚ　＝　１００

    求ｘ，ｙ，ｚ

    代码在ｃｏｄｅｂｌｏｃｋｓ１７．１２运行的结果为

    0 25 75
    4 18 78
    8 11 81
    12 4 84

    #include <iostream>
#include <cmath>

using namespace std;

int main()
{
       float x, y, z;
       for(x=0; x<15; x++)　//７ｘ　＋　４ｙ　＝　１００（可以推出）
      {
             y = (100-7*x)/4;
             if(floor(y)==y)　　//先对ｙ取整数部分，并判断与原值是否相等，
                                            //这里相当于取出整数ｙ值
            {
                   z = 100-x-y;
                   cout<< x <<' '<<y<<' '<<z<<endl;
            }
       }
       return 0;
}
