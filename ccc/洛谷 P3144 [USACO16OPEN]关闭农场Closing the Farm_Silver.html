    题目

    P3144 [USACO16OPEN]关闭农场Closing the Farm_Silver

    思路

    先全部读入利用并查集倒着处理

    \(Code\)

    #include<iostream>
#include<cstring>
#include<cstdio>
#include<string>
#include<algorithm>
#define MAXN 3001
using namespace std;
int n,m;
bool vis[MAXN];
int fa[MAXN],r[MAXN],ans[MAXN];
int u[MAXN],v[MAXN],del[MAXN];
int find(int x){
    return fa[x]==x?x:fa[x]=find(fa[x]);
}
void Union(int x,int y){
    int rootx=find(x),rooty=find(y);
    if(rootx==rooty) return;
    if(r[rooty]>r[rootx]) fa[rootx]=rooty;
    else if(r[rootx]>r[rooty]) fa[rooty]=rootx;
    else{
        fa[rooty]=rootx;
        r[rootx]++;
    }
}
inline void read(int &T){
    int x=0;bool f=0;char c=getchar();
    while(c<'0'||c>'9'){if(c=='-')f=!f;c=getchar();}
    while(c>='0'&&c<='9'){x=x*10+c-'0';c=getchar();}
    T=f?-x:x;
}

int main(){
    read(n),read(m);
    for(int i=1;i<=m;++i) read(u[i]),read(v[i]);
    for(int i=1;i<=n;++i) read(del[i]),fa[i]=i;
    for(int i=n;i>0;--i){
        vis[del[i]]=1;
        for(int j=1;j<=m;++j){
            if(vis[u[j]]&&vis[v[j]]){
                Union(u[j],v[j]);
            }
        }
        for(int j=1;j<=n;++j){
            if(vis[j]&&find(j)==j){
                ans[i]++;
            }
        }
    }
    for(int i=1;i<=n;++i){
        if(ans[i]==1) puts("YES");
        else puts("NO");
    }
    return 0;
}
