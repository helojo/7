    利用二分查找算法和C++短路求值特性。

    平台AC代码：

     1 class Solution {
 2 public:
 3     int searchInsert(vector<int>& nums, int target) {
 4         int lo = 0, hi = nums.size();
 5         while (lo < hi){
 6             int mi = lo + ((hi - lo) >> 1);
 7             (target < nums[mi]) ? hi = mi : lo = mi + 1;
 8         }
 9         int i = --lo;
10         if (i == -1 || nums[i] != target) i++;
11         return i;
12     }
13 };
