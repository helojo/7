    NO.41 P2676 超级书架

    #include<iostream>
    #include<algorithm>
    using namespace std;
    int main()
    {
        int i,n,b,ans,m;
        int a[100001];
        cin>>n>>b;
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        sort(a,a+n);
        i=n;
        ans=0;
        m=0;
        while (m<b)
        {
            m+=a[i--];
            ++ans;
        }
        cout<<ans-1<<endl;
        return 0;
    }

    scanf("%d",&n);

    a[i] = 0;
    a[1] = 1;
    }
    void get_ans()
    {
    for (int i = 1;i <= n;i++)
    {
    int x = 0;

    a[j] = a[j] *i + x;
    x = a[j] / 10;
    a[j] = a[j] % 10;
    }
    while ( x > 0)
    {
    l1++;
    a[l1] += x;
    x = a[l1] / 10;
    a[l1] = a[l1] % 10;
    }
    }
    }
    void output_ans()
    {
    for (int i = l1;i >= 1;i--)
    printf("%d",a[i]);
    }
    int main()
    {
    input_data();
    get_ans();
    output_ans();
    return 0;
    }
