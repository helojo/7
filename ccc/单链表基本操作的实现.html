     3.插入节点

    Status insertList(LinkList L,int local,ElemType &e)
{
    if(local<1||local>(L->data+1))
    {
        printf("invalid input\n");  //判断插入的未知是否有效
        return FALSE;
    }

    LinkList p = L;
    for(int i=1;i<local;i++)
    {
        p = p->next;       
    }

    LinkList s = new Lnode;

    s->data = e;
    s->next = p->next;
    p->next = s;

    L->data++;           //插入后长度加1
    return TRUE;

}
