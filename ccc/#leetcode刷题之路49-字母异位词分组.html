    给定一个字符串数组，将字母异位词组合在一起。字母异位词指字母相同，但排列不同的字符串。
    示例:
    输入: ["eat", "tea", "tan", "ate", "nat", "bat"],
    输出:
    [
    ["ate","eat","tea"],
    ["nat","tan"],
    ["bat"]
    ]
    说明：
    所有输入均为小写字母。
    不考虑答案输出的顺序。

     

    思路：建立map，放就完了

    #include <iostream>
#include <vector>
#include <algorithm>
#include <map>
using  namespace std;

vector<vector<string>> groupAnagrams(vector<string>& strs) {
    vector<vector<string>> ans;
    if (strs.empty()) {
        return ans;
    }
    map<string, vector<string>> mp;
    for (string s : strs) {
        string tmp = s;
        sort(tmp.begin(), tmp.end());
        mp[tmp].push_back(s);
    }
    //for (pair<string, vector<string>> a : mp) {
    for (auto a : mp) {
        //vector<string> t(a.second.begin(), a.second.end());
        vector<string> t=a.second;
        ans.push_back(t);
    }
    return ans;
}

int main() {
    vector<string> a={"eat", "tea", "tan", "ate", "nat", "bat"};
    vector<vector<string>> ans=groupAnagrams(a);
    std::cout << ans.size() << std::endl;
    return 0;
}
