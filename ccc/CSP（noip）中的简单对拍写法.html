    以a+b为例

    这是随机数据

    #include<iostream>
#include<cstdio>
#include<ctime>
using namespace std;
int main(){
    freopen("rand.txt","w",stdout);
    int a,b;
    srand(time(0));
    a=rand()%10+1;b=rand()%10+1;
    cout<<a+b;
}

    这是正解（假设它是正解！）

    #include<iostream>
#include<cstdio>
#include<ctime>
using namespace std;
int main(){
    freopen("rand.txt","r",stdin);
    freopen("1.txt","w",stdout);
    int a,b;
    cin>>a>>b;
    cout<<a+b;
}
