    R与r的大小，多种情况进行判断

    看题的时候，对数据的严谨性

    #include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define PI 3.1415926

int main()
{
    double n, R, r;
    while(~scanf("%lf %lf %lf", &n, &R, &r))
    {   
        if (r > R)
        {
            cout << "NO" << endl;
            continue;
        }
        else if(r == R && n == 1)
        {
            cout << "YES" << endl;
            continue;
        }
        else if(r > R/2 && n == 1)
        {
            cout << "YES" << endl;
            continue;
        }
        else if(r == R/2 && n == 2)
        {
            cout << "YES" << endl;
            continue;
        }
        else
        {
            double a = PI / asin(r / (R - r));
            if (a > n)
            {
                cout << "YES";
            }
            else
            {
                cout << "NO";
            }
            cout << endl;
        }
    }
    return 0;
}
