14 //类
15 class TEST{
16     int DATA[3][3];
17 public:
18     void function1(char a)
19     {
20         cout<<a<<endl;
21         //在类内定义成员函数
22     }
23     void function2(char a);
24     /*
25         注意！ 在类外定义成员函数
26         必须先在类体中作原型声明，然后
27         在类外定义
28     */
29 };
30 
31 void TEST::function2(char a)
32 {
33     cout<<a<<endl;
34     //在类外定义成员函数
35 }

41     cout<<s<<endl;
